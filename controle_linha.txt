	luz_verde : BOOL;        // Luz indicadora de operação normal

	botao_start_anterior : BOOL;
END_VAR	

// LÓGICA PRINCIPAL

    sistema_ligado := FALSE;
    luz_vermelha := TRUE;
    luz_verde := FALSE;
    luz_amarela := FALSE;
    motor_esteira := FALSE;
    alarme := TRUE;
ELSE
    alarme := FALSE; // Alarme desliga ao liberar a emergência
END_IF;

ELSE

    IF botao_start AND NOT botao_start_anterior THEN
        sistema_ligado := TRUE;
    END_IF;
    
    // Lógica de parada
    IF botao_stop THEN
        sistema_ligado := FALSE;
    END_IF;
END_IF;

// Monitoramento de temperatura ( adicionado na manutenção )

 	  sistema_ligado := FALSE ;
 	END_IF ;
   ELSE
 	luz_amarela := FALSE ;
 	alarme := FALSE ;
   END_IF ;

    motor_esteira := TRUE;
    luz_verde := TRUE;
    luz_vermelha := FALSE;
ELSE
    motor_esteira := FALSE;
    luz_verde := FALSE;
    luz_vermelha := TRUE;
END_IF;

// Contagem de peças (borda de subida)
IF sensor_fim AND NOT sensor_fim_anterior THEN
    contador_pecas := contador_pecas + 1;
END_IF;
